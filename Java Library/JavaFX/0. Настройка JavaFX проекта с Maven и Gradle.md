[[JavaFX]]

Maven
Код: `pom.xml`
``` xml ignore
<?xml version="1.0" encoding="UTF-8"?>
<project xmlns="http://maven.apache.org/POM/4.0.0" 
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 
                             http://maven.apache.org/xsd/maven-4.0.0.xsd">
                             
    <modelVersion>4.0.0</modelVersion>
    <groupId>com.example</groupId> <!-- Идентификатор группы проекта -->
    <artifactId>javafx-app</artifactId> <!-- Идентификатор артефакта проекта -->
    <version>1.0.0</version> <!-- Версия проекта -->
    <packaging>jar</packaging> <!-- Тип упаковки -->
    
    <properties>
        <java.version>17</java.version> <!-- Версия Java -->
        <javafx.version>20.0.2</javafx.version> <!-- Версия JavaFX -->
    </properties>

    <dependencies>
        <!-- Основные модули JavaFX -->
        <dependency>
            <groupId>org.openjfx</groupId>
            <artifactId>javafx-controls</artifactId>
            <version>${javafx.version}</version>
        </dependency>
        <dependency>
            <groupId>org.openjfx</groupId>
            <artifactId>javafx-fxml</artifactId>
            <version>${javafx.version}</version>
        </dependency>
        <!-- Добавьте другие модули JavaFX по необходимости -->
    </dependencies>

    <build>
        <plugins>
            <!-- Плагин для компиляции Java -->
            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-compiler-plugin</artifactId>
                <version>3.8.1</version>
                <configuration>
                    <source>${java.version}</source> <!-- Версия исходного кода -->
                    <target>${java.version}</target> <!-- Целевая версия JVM -->
                </configuration>
            </plugin>
            
            <!-- Плагин для запуска JavaFX-приложения -->
            <plugin>
                <groupId>org.openjfx</groupId>
                <artifactId>javafx-maven-plugin</artifactId>
                <version>0.0.8</version>
                <executions>
                    <execution>
                        <goals>
                            <goal>run</goal> <!-- Цель запуска приложения -->
                        </goals>
                    </execution>
                </executions>
                <configuration>
                    <mainClass>com.example.MainApp</mainClass> <!-- Полный путь к основному классу -->
                </configuration>
            </plugin>
        </plugins>
    </build>
</project>
```

Gradle
Код: `build.gradle` (Groovy DSL)
``` groovy ignore
plugins {
    id 'application' // Применение плагина для запуска приложения
    id 'org.openjfx.javafxplugin' version '0.0.13' // Применение JavaFX плагина
}

repositories {
    mavenCentral() // Использование Maven Central для зависимостей
}

dependencies {
    // Основные модули JavaFX
    implementation 'org.openjfx:javafx-controls:20.0.2'
    implementation 'org.openjfx:javafx-fxml:20.0.2'
    // Добавьте другие модули JavaFX по необходимости
}

javafx {
    version = "20.0.2" // Версия JavaFX
    modules = [ 'javafx.controls', 'javafx.fxml' ] // Модули JavaFX, которые используются
}

application {
    mainClass = 'com.example.MainApp' // Полный путь к основному классу
}

java {
    sourceCompatibility = JavaVersion.VERSION_17 // Версия исходного кода
    targetCompatibility = JavaVersion.VERSION_17 // Целевая версия JVM
}
```
