[[JavaBase]]
class Demo{
	public static void main(String[] args){
%%Индексная переменная и верхняя граница суммы%%
		int i,n=100;
%%Переменная для записи значения суммы%%
		int sum=0;
%%Оператор цикла%%
		for(i=1;i<=n;i++){
			sum+=i;
		}
		System.out.println(
			"Сумма чисел от 1 до "+n+": "+sum
		);
	}
}
